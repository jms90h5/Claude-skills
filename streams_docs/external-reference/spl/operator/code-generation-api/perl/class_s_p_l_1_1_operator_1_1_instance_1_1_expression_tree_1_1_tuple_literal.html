sub toString
{
my ($self) = @_;
my $kind = $self->getKind();
my $ret = '{';
my $numElements = $self->getNumberOfElements();
for (my $i = 0; $i < $numElements; $i++) {
$ret .= ", " if $i > 0;
$ret .= $self->getAttributeAt($i);
$ret .= '=';
$ret .= $self->getLiteralAt($i)->toString();
}
return $ret . '}';
}
