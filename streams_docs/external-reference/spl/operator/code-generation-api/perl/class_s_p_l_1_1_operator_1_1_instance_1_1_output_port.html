sub getSchema($$)
{
my ($self, $rootName) = @_;
if ($self->{xml_}->{tuple}) {
my $xml = XML::Simple->new (ForceArray => 1, KeepRoot => 0, RootName => $rootName, keyattr => []);
return $xml->XMLout ($self->{xml_}->{tuple}->[0]);
}
return undef;
}
